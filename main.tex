% Marca de agua controlada por capítulo y clase
% Para poner el watermark de DRAFT. 
% Las macros son para poder poner y sacar en cada capítulo a medida que se vayan terminando
\usepackage{draftwatermark}
% Default: no watermark. Chapters can enable it with \draftwatermarkon
\SetWatermarkText{}
\SetWatermarkScale{1}

\newcommand{\draftwatermarkon}{%
	\SetWatermarkText{DRAFT}%
	\SetWatermarkScale{1}%
	\gdef\draftwatermark@text{DRAFT}%
	\gdef\draftwatermark@scale{1}%
}
\newcommand{\draftwatermarkoff}{%
	\SetWatermarkText{}% empty => no watermark
	\gdef\draftwatermark@text{}
}

\begin{document}
\draftwatermarkoff

% Portada
\input{caratula}

% Capítulos (control centralizado del watermark desde main.tex)
% Ejemplo: activar watermark para el capítulo 1 y apagar para el resto
% Use per-chapter control files so a single file controls both
% compilación individual y compilación general.
\clearpage
% Control centralizado de watermark: activar antes del capítulo 1
\draftwatermarkon
\subfile{chapters/chapter-1}
\clearpage
% Apagar a partir del capítulo 2
\draftwatermarkoff
\subfile{chapters/chapter-2}
\clearpage
\draftwatermarkoff
\subfile{chapters/chapter-3}
\clearpage
\draftwatermarkoff
\subfile{chapters/chapter-4}
\clearpage
\draftwatermarkoff
\subfile{chapters/chapter-5}
